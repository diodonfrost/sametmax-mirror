>>> import ctypes
>>> data = [1, 2, 3]
>>> # création d'un array d'int de taille 3
>>> TypeArray = ctypes.c_int * len(data)
>>> array = TypeArray(*data)
>>> array
<__main__ .c_int_Array_3 object at 0x7ff84f9f3e18>
>>> array[0]
1
